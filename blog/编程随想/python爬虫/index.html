from bs4 import BeautifulSoup
# import json
response = requests.get(url,headers=header)
# json = json.loads(response.content)
text = BeautifulSoup(response.text,'html.parser')  #将html页面格式化/补齐标签
# print(text.prettify())   #美化/换行
a_tags = text.find_all('a','mnav')   #获取a标签里class='mnav'的语句
for a in a_tags:
	print(a.get('href'))    #获取标签内href的值
	print(a.text)    #获取标签内部字符串
