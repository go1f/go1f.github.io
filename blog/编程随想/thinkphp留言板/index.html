ThinkPHP MVC模型探索
编程语言 PHP
Lastmod: 2020-03-06

THINKPHP3.2.3 基于MVC的模型，迎合前后端分离的潮流，加上详尽的开发者手册，国内较为受欢迎的PHP 开发框架。现在出到5.x了，但3.2.3是个经典。MVC模型在大三下的时候就接触到了，当时是看着ASP.NET去学的。

看着哔哩哔哩的视频、官方文档半懂不懂，又要赶出设计作业交，有点浮躁，最后感觉做的有点糊涂。回想要补坑，挖洞时候，也经常看见这个模型，就想着实践一下去了解应用，也是对开发者的一点体验，虽然只是做 Hello World 式的留言板。

数据库设计 设计留言表与用户表

​

ThinkPHP3.2.3

默认三个文件夹 Application、Public、ThinkPHP，ThinkPHP文件夹是框架的库，Application就是开发的项目存放。

着重看Application目录，标志性的Controller、Model、View目录，Common目录存放公共函数，是框架运行前加载的文件。Conf则是配置的参数，如数据库的登录账号等。

控制器类文件要求以控制器名+Controller.class.php格式命名，譬如这里命名了Index与User的类，分别打算处理留言与账号的功能。

​

Model

Model是指数据库的设计模型，跟Controller做数据交互。

<?php
#code: MessagesModel.php 
namespace Home\Model;
use Think\Model;
class MessagesModel extends Model {
    protected $trueTableName = 'messages';
}

简单定义类就可以了。关键是添加数据库的配置。

调用时候，用ThinkPHP的M函数，会自动读取配置文件，新建出对应的模型对象。再调用模型的方法完成增删改查。

​

View

以展示留言这个功能，介绍前后端数据的传送。默认的模板文件定义规则是：

视图目录/[模板主题/]控制器名/操作名+模板后缀

Index控制器类的 show_message 方法，从model模型取出数据，放到 $message_list，调用assign，指定变量与前端调用的接口名。

View/Index/index.html 调用命名为message_list的变量就可以了。

前端，使用了循环列表标签，二维数组的取值。

​

U函数调用，前端添加{: }，比如调用User控制器login方法，则{:U('/User/login')}

后端的跳转同理，配合sucess、error方法跳转

​

Controller

定义的方法，供调用而已，如前文提到的login、show_message

还有常用到的 I函数，用于读取请求中的GET、POST数据，并作过滤（防止SQL注入、XSS）

​

Controller

IndexController.class.php

<?php
namespace Home\Controller;
use Think\Controller;
class IndexController extends Controller {
    public function index(){
        $this->show_message();
        $this->show();
        log_access();
    }

    public function show_message(){
        $messagesModel = M("messages");
        $message_list = $messagesModel->order("create_time desc")->page(1,8)->select();
        $this->assign('message_list',$message_list);
        // dump($message_list);
    }

    public function add(){
        $this->show();
    }

    public function do_add(){
        $this->check_login();
        $content = I("content");
        if(empty($content)){
            $this->error("内容不能为空！");
        }
        $messagesModel = M("Messages");
        $data = array("user_id"=>session("user_id"),"username"=>session("username"),"content"=>$content,"create_time"=>date('Y-m-d')  );
        if($messagesModel->create($data) && $messagesModel->add() ){
            $this->success("留言成功！", U("Index/index"));
        }
        else{
            $this->error("留言插入失败！！");
        }
    }

    public function check_login(){
        if( !session("user_id") ){
            $this->error("请先登录！", U("User/login"));
        }
        else{
            // $this->sucess("已登录！")
        }
    }
}

UserController.class.php

<?php
namespace Home\Controller;
use Think\Controller;
class UserController extends Controller {
    public function index(){
        $this->show('Hello Login!');
    }

    public function regesiter(){
        $this->show();
    }

    public function login(){
        $this->show();
    }
    
    public function logout(){
        $this->do_logout();
    }
    public function do_register(){
        if(session()){
            $this->error("你已经登录！", U("Index/index"));
        }
        $username = I("username");
        $password = I("password");
        $password_repeat = I("password_repeat");
        if( !($username && $password && $password_repeat) ){
            $this->error("请输入账号或密码！！",U("User/login"));
        }
        if($password_repeat != $password){
            $this->error("密码不一致！！",U("User/register"));
        }
        
        $userModel = M("Users");
        $password = md5($password."go1f223");
        $data = array("username"=>$username,"password"=>$password);
        $result = $userModel->where($data)->find();
        if($result){
            $this->error("该用户已经注册！！", U("User/register"));
        }
        $userModel = M("Users");
        $data = array("username"=>$username,"password"=>$password);
        if( $userModel->create($data) && $userModel->add() ){
            $result = $userModel->where($data)->find();

            session('user_id',$result["id"]);
            session("username",$result["username"]);
            log_register_sucess();
            
            $this->success("注册成功！", U("Index/index"));
        }
        else{
            $this->error("注册失败，请稍后再试。", U("Index/index"));
        }
    }

    public function do_login(){
        if(session()){
            $this->error("你已经登录！", U("Index/index"));
        }
        $username = I("username");
        $password = I("password");
        if( !($username && $password) ){
            $this->error("请输入账号或密码！！",U("User/login"));
        }

        $userModel = M("Users");
        $password = md5($password."go1f223");
        $data = array("username"=>$username,"password"=>$password);
        $result = $userModel->where($data)->find();

        if(empty($result)){
            log_login_attempt($username, $password);
            $this->error("账户名或密码错误！！");
        }

        session('user_id',$result["id"]);
        session("username",$result["username"]);
        log_login_sucess();
        
        $this->success("登录成功！", U("Index/index"));

    }

    
    public function do_logout(){
        log_logout();
        session_destroy();
        $this->success("退出成功！", U("Index/index"));
    }

}

Common

Common/function.php 主要实现日志记录

<?php

    function log_login_sucess(){
        $fp = fopen("log.txt","a") or die("Unable to open file!");
        $data = date("Y-m-d h:i:sa")." login in, uname=".session("username") .", id=".session("user_id").", ip=".$_SERVER["REMOTE_ADDR"]."\n";
        fwrite($fp,$data);
        fclose($fp);
    }

    function log_login_attempt($username,$password){
        $fp = fopen("log.txt","a+")  or die("Unable to open file!");
        $data = date("Y-m-d h:i:sa")." login failed, uname=".$username." with pw=".$password. " ip=".$_SERVER["REMOTE_ADDR"]."\n";
        fwrite($fp,$data);
        fclose($fp);
    }

    function log_access(){
        $fp = fopen("log.txt","a+")  or die("Unable to open file!");
        $data = date("Y-m-d h:i:sa")." access, ip=".$_SERVER["REMOTE_ADDR"]."\n";
        fwrite($fp,$data);
        fclose($fp);
    }

    function log_logout(){
        $fp = fopen("log.txt","a+")  or die("Unable to open file!");
        $data = date("Y-m-d h:i:sa")." login out, uname=".session("username")."\n";
        fwrite($fp,$data);
        fclose($fp);
    }

    function log_register_sucess(){
        $fp = fopen("log.txt","a") or die("Unable to open file!");
        $data = date("Y-m-d h:i:sa")." register, uname=".session("username") .", id=".session("user_id").", ip=".$_SERVER["REMOTE_ADDR"]."\n";
        fwrite($fp,$data);
        fclose($fp);
    }
